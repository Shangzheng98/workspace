Structure about USB3 device information

Structure Definition

struct{
  unsigned char     CrtlInEndPoint;

  unsigned short    idVendor;     
  unsigned short    idProduct;

  unsigned char     chDeviceGUID[INFO_MAX_BUFFER_SIZE/*64*/];
